### How it work?

```ts
import { Component, EventEmitter, Input, Output, ViewEncapsulation } from "@angular/core";
import { CommonModule } from '@angular/common';
import { OffcanvasComponent, OffcanvasPosition } from "@daibh/material";

@Component({
  selector: 'storybook-offcanvas-example',
  template: `
    <div class="wrapper">
      <nw-offcanvas #offcanvas [title]="title" [position]="position" [backdrop]="backdrop" [(opened)]="opened">
        <div>
          <h3>demo offcanvas content</h3>
        </div>
      </nw-offcanvas>
      <button class="btn btn-sm btn-primary ms-3" (click)="open()">Open offcanvas</button>
      <button class="btn btn-sm btn-primary ms-3" (click)="offcanvas.open()">Open offcanvas by ref</button>
      <button class="btn btn-sm btn-primary ms-3" (click)="openByService()">Open offcanvas by service</button>
    </div>
  `,
  styleUrl: './offcanvas-example.component.scss',
  standalone: true,
  imports: [
    OffcanvasComponent
  ],
  providers: [
    { provide: NW_OFFCANVAS_CONFIG, useValue: { name: 'offcanvas-example', backdrop: true } }
  ],
  encapsulation: ViewEncapsulation.None
})
export class OffcanvasExampleComponent {
  private readonly _offcanvasService: OffcanvasService = inject(OffcanvasService);

  @Input() title: string = 'Demo offcanvas';
  @Input() position: OffcanvasPosition = 'start';
  @Input() opened: boolean = false;
  @Output() openedChange = new EventEmitter<boolean>();

  open(): void {
    this.opened = true;
    this.openedChange.emit(this.opened);
  }

  openByService(): void {
    this._offcanvasService.nextState('offcanvas-example', true);
  }
}
```
